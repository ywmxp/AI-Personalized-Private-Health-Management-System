### 1.2 request.ts 核心设计

```TypeScript
// Axios 实例创建
const request = axios.create({
  timeout: 30000,  // 30 秒超时（适应 AI 接口较长响应时间）
})

// 请求拦截器
request.interceptors.request.use((config) => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器
request.interceptors.response.use(
  (response) => {
    // 业务级错误处理（code !== 0）
    // 特殊处理 code 50000 → AI 服务异常
  },
  (error) => {
    // HTTP 级错误处理
    // 401 → 清除 token + 跳转登录
    // 超时 → 特殊提示
  }
)
```

## 2. 统一响应格式

### 2.1 ApiResponse<T>

```TypeScript
/** 后端统一响应格式 */
interface ApiResponse<T = unknown> {
  code: number       // 0 = 成功，非 0 = 业务错误
  message: string    // 提示信息
  data: T            // 业务数据
}
```

### 2.2 分页响应

```TypeScript
interface PageData<T> {
  items: T[]      // 数据列表
  page: number     // 当前页码
  size: number     // 每页条数
  total: number    // 总记录数
}
```

### 2.3 错误码约定

|code|含义|前端处理|
|-|-|-|
|0|成功|正常返回|
|非 0|业务错误|`ElMessage.error(data.message)`|
|50000|AI 服务异常|精简提示"AI 服务异常，请稍后重试"|
|—|HTTP 401|清除登录状态，跳转 `/login`|
|—|HTTP 超时|"请求超时，AI 正在处理中，请稍后重试"|
|—|网络错误|`ElMessage.error('网络错误')`|

## 4. 接口调用规范

### 4.1 View 层调用规范

```TypeScript
// ✅ 正确：通过 api/ 层调用
import { getHealthRecords } from '../api/health'
const res = await getHealthRecords({ page: 1, size: 10 })

// ❌ 错误：直接在 View 中 import axios
import axios from 'axios'
const res = await axios.get('/api/health-data')
```

### 4.2 错误处理规范

```TypeScript
// 拦截器已统一处理通用错误（网络错误、401、超时）
// View 中只需处理特定业务逻辑：

async function fetchRecords() {
  try {
    const res = await getHealthRecords({ ... })
    records.value = res.data.data.items
  } catch (err: unknown) {
    // 仅处理需要特殊逻辑的错误（如 AI 画像 50000 无数据提示）
    const e = err as Error & { code?: number }
    if (e.code === 50000) {
      noDataError.value = true
    }
    // 通用错误已在拦截器中 toast 提示，无需重复处理
  } finally {
    loading.value = false
  }
}
```

### 4.3 Mock 分支规范

```TypeScript
// 每个 API 函数内部判断 Mock 开关
export async function getHealthRecords(params?) {
  if (isMockEnabled()) {
    return mockSuccess(getMockHealthRecords())
  }
  return request.get('/api/health-data', { params })
}

// 调用方零改动：Mock 和真实请求返回格式完全一致
```

## 6. 请求/响应时序（典型流程）

### 6.1 登录流程

```Mermaid
sequenceDiagram
    participant V as LoginView
    participant A as auth.ts
    participant R as request.ts
    participant BE as Spring Boot
    participant S as userStore
    participant LS as localStorage

    V->>A: login(phone, password)
    A->>R: request.post('/api/auth/login', {...})
    R->>R: 请求拦截器（无 token，跳过）
    R->>BE: POST /api/auth/login
    BE-->>R: { code: 0, data: { token, user } }
    R->>R: 响应拦截器（code=0，通过）
    R-->>A: response
    A-->>V: { data: { code, data: { token, user } } }
    V->>S: setLoginInfo({ token, user })
    S->>LS: 持久化
    V->>V: router.push('/home')
```

### 6.2 AI 生成画像流程

```Mermaid
sequenceDiagram
    participant V as AiHealthProfile
    participant A as ai.ts
    participant R as request.ts
    participant BE as Spring Boot → FastAPI

    V->>V: 设置 generating=true + 显示 LoadingSpinner
    V->>A: generateProfile({ timeRange })
    A->>R: request.post('/api/ai/profile', {...})
    R->>R: 请求拦截器附加 Authorization
    R->>BE: POST /api/ai/profile
    Note over BE: AI 分析处理中（可能 5-10 秒）
    BE-->>R: { code: 0, data: AiProfileData } | { code: 50000, ... }
    R-->>A: response
    A-->>V: ApiResponse
    V->>V: 设置 generating=false + 渲染结果
```

## 7. Vite 代理配置

```TypeScript
// vite.config.ts
server: {
  host: '127.0.0.1',
  port: 5173,
  proxy: {
    '/api': {
      target: 'http://127.0.0.1:8080',   // 后端 Spring Boot
      changeOrigin: true,
    },
  },
}
```

**说明：**

- 开发环境中，前端所有 `/api/*` 请求被 Vite 代理到后端

- 生产环境由 Nginx 等反向代理统一处理

- 前端代码中使用 `/api/xxx` 相对路径，无需区分环境
